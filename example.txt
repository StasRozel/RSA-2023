number function min(number x, number y)
{
	variable: number res;
	if: x < y #
	istrue {res = x;}
	isfalse {res = y;} #
	return res;
}

script function strings (string a, string b)
{
	variable: number k;
	k = lenght(a) + 5;
        write "Lenght + 5:          ";
	writeline k;
	
	a = strcopy(b);
        write "String copy:         ";
	writeline a;
	return;
}
